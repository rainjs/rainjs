Class ConsoleAppender¶

Colored console appender:

{
    "type": "console",
    "layout": {
        "type": "pattern",
        "params": {
            "pattern": "[%level] %date: %message"
        }
    },
    "params": {
        "debug": {
            "foreground": "green"
        },
        "info": {
            "foreground": "cyan"
        },
        "warn": {
            "foreground": "yellow"
        },
        "error": {
            "foreground": "red"
        },
        "fatal": {
            "foreground": "black",
            "background": "red"
        }
    }
}
